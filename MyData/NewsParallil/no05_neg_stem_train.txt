ئىنسان بۇنچە ۋەھشىي بولار بىل ئىنسان بولۇپمۇ بىز گومىنداڭ ئەسكەر ۋەھشىيلىك ھەرقانداق يىرتقۇچ باس چۈش
ئاتىلىق ئۇرۇق ئانىلىق تۇخۇم ياخشى بول ساغلام ياكى تازا بول سۇس بول تۈپەيلىدىن بالا شەكىل جەھەت ھەرخىل يىتەرسىزلىك بىلەن تۇغ
دەرۋەقە ئۇ ئالقان ئارقىلىق ئايان بول شەپقەتسىز قىسمەت تېخى ئاغزى ئانا سۈت پۇرا تۇر بىر بۆدەر تەقدىر پىشانە پۈتۈلگەن ئاغىرىنغان چوڭقۇر تىن ئۇھ تارت
ئۇ تۇنجى توي ئاخشام ئون بەش ياشلىق بۇ باي قىز زوقمەنلىك كۆيۈك ئىچى مۇشۇنداق زوقلان قارا
ئادەملەر قورق چەكچەي قال
ھېلىقى لۈكچەك لەرنىڭ ئىككىسى ئۇنىڭ يان دە پاراڭ سال تۇر بۇ لەنىتى تېخى ئۇخلا ؟
خاتىپ مىراب ئالدىرا ئاغزى چىق گەپ بىلەن ھېززىبەگ ھاجى تارت پۈتۈن مەكىت مۆتىۋەر چەكسىز ئەيىبلە نەزىرى ئۇنىڭ چەكچەي
سۈكۈت قىل يولسىزلىق شېرىكچىلىك ۋە ئۇنىڭ قارشى كۈرەش جۈرئەتسىزلىك ئىپادىلىندىغان ۋىجدانسىزلىق
ئۇنىڭ ئەلپاز ھىيلىگەرلىك چىق تۇر بۇ تاغار قانداقمۇ يۆتكە بول بولا ؟
نېمە بۇنچىۋالا تەشۋىش قال ؟
بەزى بەزىدە تۆگە كولدۇرما پەرىشان جاراڭلا كەت ئاڭلا قال
ئەسكەر روھسىز جەڭگىۋارلىق ئاجىز
ئۇ شۇ قەدەر ياۋاش بول شۇ تۈپەيلىدىن ئادەم بىر ئاز كۈلكىلىك تۇيۇل
چېرىك ئەسەبىي ئۇ قامچىلا باشلا
بەزى ھەسەت ۋە كۆرەلمەسلىك ئۆز بۇرادەر سات ۋە ياكى ئۇنىڭ بىۋاسىتە قول سال
ساپاسىزلىق مەسئۇلىيەتسىزلىك تۈپەيلى ئەۋلاد بىز قول نابۇت بول
كىشى ئۆز تۇرمۇش كەسپ ئاغرىنىپ يۈر ئۆز بىرەر ئىش ۋۇجۇد چىقىرالماسلىق سەۋەب ياش چوڭاي قال دۆڭگە قوي
يۈزلىگەن كىشى گۇمان بىلەن تۈرمە تاشلا
ۋاي ئىسىت بىر قىسم ئىش بول كەت مىلتىق ئايرى پالگان باش بۇنىڭ ئاياغ نە چىق ؟
ئۇلار ئاچ غالىلدا تىترە ئىد
ئايخان بىردىنلا تەشۋىش تۇر قال ئۇنىڭ كۆڭۈل گۇمان چۈش بول
ئۇنىڭ دۈگىلەك يۈز قانداقتۇر بىر غەم ئەندىشە ئالامەت ئەكس ئات تۇر
رەھمەتلىك قاسىم سەيدى تولىمۇ ئىقتىدارلىق كىشى بولغاچقا نۇرغۇن ھەسەتخور ئۇنىڭ دۈشمەنلىك قىل ئۇنىڭ ئۆس يول توس قوي
ئۇلار خۇدا قورق مۇنداق ناھەق قان تۆك ھەرگىز رازى ئەمەس
گۈلھېپىزە كۆز دەسلەپ ئاجىز بارا بارا كۈچلۈك ئەسەبىي بىر قاينام ھاسىل بول
دە ھېلىقى سۈرلۈك ئادەم ئاچچىق بىلەن ئاڭلا قوي خالايىق كىم كىم گەدەنكەشلىك قال بىز يان ئۇن باغلا ئاس سولا ئۇر
ئاشۇنداق روھىي ھالەت ئاستىدا ئۇ ئۆز بىئارام بول ھېس قىل شۇ بىئاراملىق بەدەن ئايان بول دە ؟
يىگىت قەلبى ئۆز جۈرئەتسىزلىك ئەيىبلە بىر خىل سۇس پۇشايمان قاپلا ئىد
رىئال ئەھۋال قارا كۆپ مىقدار نامۇۋاپىق تەرتىپسىز ئاچ ھەرىكەت ۋە ئىشلەپچىق پائالىيەت سەۋەب قىممەتلىك دۆلەت زىمىن ياخشى پايدىلان ۋە قوغدا ئېرىش
ئوغۇل دادا توخۇ يۈرەك ئىدىيە كونا يەككە دېھقان ئىدىيى كۈچلۈك دە ئەيىبلە باشلا ياسا تۇر بۇز قانداق بىل ؟
ئىدىرىس چىراي ئالاقزادىلىك قورقۇنچ تول
ۋاڭ گۇرۇڭ بۇنىڭ ئىنتايىن ئازابلا تەقدىر تەڭسىزلىك ئاغرىن
چۈنكى ھىيلىگەرلىك تۈلكە قال سىلۋېر كېمە ئالتە ئادەم قال قوي
ئۇنىڭ يوغان بۇرۇن غەزەپ مىدىرلا يوغان كۆز قويۇق قاش ئاستىدا پالىلدا قويۇق ساقال بۇرۇت بىلىنەر بىلىنمەس تىترە
ئادەم ئاساس قىل ئەڭ ئەقەللىي تەلەپ ئادەم ھايات كۆيۈن ئادەم ساغلاملىق ئەتىۋار ئىبارەت
مانا بۇ جياڭ زۇڭتۇڭ ئەندىشە مېنىڭ ئەندىشە !
بەزى ئۆز زەئىپلىك ئىقتىدارسىزلىق نابابلىق تۈپەيلى بىر ئۆمۈر ئاتا ئانا يۆلەن تەمەخورلۇق ئادەتلەن
ئايتىللا يەر تىزلان ئاق كۆڭلەڭگە ئۈمۈدسىز قارا تۇر
تەگسىز خىيال نىمىشقا پەرىشانلىق ھەسسىلە كەت
زالىم يوقا ئادەم قانداقمۇ كۆڭۈل ئارام بول !
ھاجى ئاچچىق ئايال ئال بول ئۇنىڭ ھۆماي قارا
ئۇ شىنجاڭ قول كەت خەۋپ سەز
ئەمدى ئانا ئايرىل ياد يات ئۆز تۇت كۆز ياش يىپ ئۈز مارجان دومىلا چۆچۈش باشلا
