ياتقان يىرى جەننەت بۇلغاي ئامىن
ئېسىل فىلىم جۇمۇ
كۈنجۈت ئېسىل دورىكەن ئەمەسمۇ
شۇنداق بولسا ناھاييتى ياخشى يىزىلغان شېئىر بولۇپتۇ
مۇبارەك بولسۇن قالتس بوپتۇ
توختاتقىلى بولسا پويىزنى توختۇتۇپ ھەممىسىنى قۇتقۇزغان ياخشى بىر جان بولسىمۇ جان ئەمەسمۇ ئۇ
ناتونۇش ئادالەتگۈل نىڭ توغۇلغان كۈنىگە مۇبارەك بولسۇن
ھېسسىيات كۈچلۈك ئىپادىلىنىپتۇ
رەھمەتلىك چوڭ ئاپام مۇشۇنداق شاپاق دوپپا تىكىپ ساتاتتى بەك ئۇستا ئىدى بىر قاراپلا خۇددى ئۆلچىۋالغاندەكلا دەل قىلىپ تىكەلەيتتى
بۇنىڭمۇ ئايىقى ياراشقان چېغى مادرىتقا
فىلىمدىكى ھەقىقىي ئىسلام دىنى تېررورلۇققا قارشى تۇرىدۇ دېگەن بىر جۈملە سۆز كىشىنى ھەقىقەتەن تەسىرلەندۈرىدۇ
مىنىڭچە ئادىداس كۈچلۈكتەك تۇرىدۇ ھەرقايسى جەھەتتىن
ئاتىلارغا بىغىشلىغان بۇ شېئىرلىرىڭىزدىن مەمنۇن بولدۇم داۋاملىق يوللاپ تۇراسىز
بۇ خەۋەر تازا قۇلاققا ياقمىدى لىكىن قاملاشتۇرۇپسىز
خېلى كەلتۈرۈپسەنغۇ مۇنبەردە ھېيتلىق بىرەر ئىش قىلساق بولاتتى
بۇ نومۇر ماڭىمۇ لازىم ئىدى ياخشى ئويلاپسىز بۇرادەر
بىز مەستانىلەر ئۈچۈنمۇ خۇش خەۋەركەن
ھەربىر ئاتائانا ئۆز پەرزەنتى ئۈچۈن ئۆرنەكتۇر
بۇ پەقەت كىتاب ئوقۇشلا ئەمەس كىشىلەردە ئۆزىنى سورايدىغان ئاڭنى يېتىلدۈرىدىغان ياخشى ئىشكەنتۇق
ئۇنداقتا تۈركىيەدە بىر ئۈستۈن روھ باركەندە
ۋاي نىمانداق كۆپ ئوماق قىزلار بۇ
سۇرەتلەرنى ياخشى تارتىپسىز سۇرەت ئوڭ تەرەپتىكى سىزغۇ دەيمەن
ياخشى بوپتۇ مەن كىتىشىنى خالايمەن
ۋاي ۋۇي نېمدىگەن مەي باغلاپ پىشقان مىۋىلەر بۇ !
كارخانچىلارنىڭ بىر بىرىنى يىتەكلەپ ئادىل رىقابەتتە ساغلام بازار ئىچىپ خەلقنىڭ كۆڭلۈدىن ئورۇن ئىلىشنى ئۈمۈت قىلىمەن
ئىسمىنىڭ بېشىغا كەلگەنلە ئىلىشقا مۇناسىپ
نادىر تىمىلىرىڭىز ئۈزۈلمىسۇن
چاخچاقلىرڭىز جايىغا چۈشۈپتۈ كۆپرەك يېزىپ تۈرۈڭ
فېستىۋال ياخشى بولدى بىراق فاينال مەننى بېسىپ چۈشكۈدەك ئىدىيە يوق تېخنىكا گۈزەللىك تەپسىلات بۇرۇنقىلارنىڭكىدىن يۇقىرى
ھايات ئۆلۈم ئۈستىدىكى زەپەر
مانا خوش خەۋەر دىگەن
بەك يا شى يىزىپسىز اللا سىزگە دەرت بەرمىسۇن مىللىتىمىزگە زادى نىمە بولدى
ھورۇنلۇقنى تاشلاپ كۆپرەك تىرىشىپ بەرسە بۇلاتتى بۇ بالا
ھاھاھاھا كويغا كاللامدەك قوغۇندىن ئى كەلدى مانا
ماۋۇ گۈدۈك ئېسل بولاپ كىتىپتۇ دەيمەن ياختۇردۇم
ئەمىلىيەتتە تۇغۇلغان كۈنىمىز ئەمەس تۇققان كۈنىڭگە مۇبارەك دېيىشىمىز كېرەك
گۈليارە غەيرەت قىلغىن
ئىتالىيلىكلەر بۇ كوستارىكا ياش چولىپىنىغا قىزىقىۋاتىدۇ قارىغاندا ئامىرىكا لوڭقىسى مۇسابىقىسى كۆپلىگەن چولپانلارنى يىتىشتۈرۈپ چىقىۋاتىدۇ
توردىن ياخشى پايدىلانسا تور بىزگە نۇرغۇن نەرسىلەرنى ئۆگتىدۇ
نەۋرۇز نامە ياخشى چىقىپتۇ
قەلەملىرىڭىزدىن مەڭگۈ تۈمەنلىگەن نۇرلار چاقنىسۇن
يەنىلا مۇشۇ قېتىم ئىشلىگىنى ياخشىكەن
چىرايى خاسىيەتلىك ئادەمگە بەك ئوخشايدۇ مەن بەك قول قويىمەن
سىزگە كۆپ رەخمەت
بىكخام مىلاندا قالغىنى ياخشى
ئاسىنال دا چاقناپ كەت
بۇ كېمە ھەقىقەتەن چوڭ ئىكەن
دەپقاپسىز جېنىم ھەممە نىمە سىزدىن ئايلانسۇن
بەللى بۇ ئىشنى شۇنداق قىل ئوغلۇم
ياخشى باشلىنىش يېرىم مۇۋەپپەقىيەت
ئۇخلىماي كۆرگەن چۈشلەر بىر بىرىدىن ياخشى چىقىپتۇ
ناخشائېيتىپ سالسا ئازاد بەگنى يېڭىۋېلىشىدا گەپ يوقكەن
تىما ئىگىسگە رەھمەت جاپا چەكتىڭىز
ئارسىنالدا ئوتتۇرا مەيدان توپچىلار ئىشىپ تىشىپ تۇرۇپتۇ ھازىرقى مۇھىم مەسىلە مىرتىساكىرنىڭ ئورنىغا قابىل ئوتتۇرا بىك ۋە كوكلىن بىلەن ماسلاشقۇدەك ئارقا بەل ۋە ئشەنچلىك ئوتتۇرا ھۇجۇمچى لازىم
گۈدۈكنىڭ كومپىيوتېر نۇسقىسى خېلى تىزلىشىپ قالدىمۇ قانداق شۇنداق خوشلۇق ئىچىدە ئىشلىتىۋاتىمەن جۇمۇ
گەپ سۆزلىرى خېلى ئورۇنلۇق جۇمۇ بۇ گاداينىڭ
بېل مادىرىدقا كەلسە نىمە دىگەن ياخشى بۇلاتتى
مۇنبەردە تىما ئاز بولغان بىلەن نادىر تىمىلار ئۈزىلمەيۋاتىدۇ
بىر كۈنلەدە بۇنىمۇ ئىشىلىتىشى مومكىن تېخنىكا ئىقتىساد ھەممىسى تەل تۇرسا ياۋرۇپادا
ئۇيغۇر خەلق ئېغىز ئەدەبىياتىنىڭ بىر تۈرى قوشاق تىن يوللىغىنىڭىزغا رەھمەت
ياخشى ئوينا دىڭ دىڭ قۇلاق
ئەلىشىرنىڭ كىنولىرىنى قەۋەتلا ياخشى كۆرىمەن
يارايدۇ ئى بىرنامبودا بىر ئىككى پاس چىقاسا بولاتتى
مەن بۇ بەتنى ياقتۇرۇپ قالدىم
قىززىغان بىرمەيدان دىبىرى
بەزى رەسىملەر چىقماپتۇ تەھرىرلىۋەتسىڭىز ئېسىل تىما بوپتۇ
بۇرۇن ئادەتتىكىلا بىر بالىتى بۇ لېكىن تۇيۇقسىزلا ئېچىلىپ كەتتىغۇ بۇ قارىۋاي ئۇزغۇنلۇق لوڭقىسىنى تالىشىش مۇسابىقىسىدىمۇ كارامەت ئوينىۋەتتى
بەك ئۇزاق يۇلغا ئاتلىنىپسەن ئىشلىرىڭ ئوڭۇشلۇق بولسۇن
تۇغۇلىغان كۇنۇڭگە مۇبارەك بولسۇن
ئىككەيلەن تەڭ چىقىشىپ قالىدۇ ھازىرقى ھالىتى
ياشاڭ ئابدۇللا ئابدۇرېھىم ئاكا ئەل دۇئاسى سىز بىلەن بىرگە
ئىلھامىڭىز ئۈزۈلمىگەي يارايسىز
سېتىلماس مال بولسا مىنىڭچە بەك ياخشى بولاتتى ھازىتمرقى بازاردىن بۇ ئىككىسىگە ئوخشاش توپچىلارنى ئۇچراتماق تەس
توغرا بەزىدە ئادەم شۇنداق بىرسىگە مۇختاج بۇلۇپ قالدۇ بەك ئەمىلي يېزلىپتۇ
بېرىپ كۆرۈشكە ئەرزىگىدەك جايكەن
مىسسى ئۇ بىر قالتىس توپچى
كومپىيوتىرىمنىڭ ئۈستەل يۈزى چىرايلىق بوپ كېتىپتۇمۇ
قالتىس كوماندا بوپتۇ باتىستۇتا مارادونا مېسسى
ەئەلەيكۇم ئەسسالام تىمىڭىزغا رەھمەت
ھازىر مۇنبىرىمىزدە پۇتبول ئاكاش باگىييولار مۇنبىرىمىزگە ھەقىقەتەن كۆپ كۈچ چىقاردى
بىكخامنىڭ چىچى ئۆزگىچىكەن
ۋاھ سۇنى سىڭگەن يەرگە سەپ
رىپىرلىك توغرىسىدا خېلى چۈشەنچىگە ئىگە بولدۇق رەخمەت
مۇشۇ شىۋە دىگەنمۇ بىر ئېسىل نەرسىكەن جۇمۇ
بۇنداق ئىشلار بۇرۇنمۇ بولغان پەيتلىك تېزلىكى دۇنيا رىكورتىنى بۇزىۋەتكەن توپچىلار بۇرۇنمۇ كۆرۈلگەن ھەممىسى قالتىس تەنھەرىكەتچىلەر مۇمكىنچىلىكى ئەلۋەتتە بار
ۋۇييۇ پەزىلەتبۇ قىزىل رەڭ سېنى نىمانچە ئاچتى ئەمدى
مۇشۇ گۈدۈكنىڭ قۇرۇلغىنىمۇ بوپتۇ سىنا ۋە باشقا خەنزۇچە بلوگلارغا ئەزا بولۇپ بېكەتكە ئۇلىغاندىن بۇنى ئۇلىساق ياخشىراق بولا
كۆپ مەزمۇن چىقماي قالدى بولمىسا مۇشۇ ئىران پۇتبولچىلىغىنى تېخىمۇ بىر تەپسىلى يېزىش خىيالىم بار ئىدى
نەپرەتنىڭ ئەستايىل ۋە مەسئۇلىيەتچانلىقىغا ھەممىمىز قايىل قارلۇق تەرەپ بۇنى ئويلىشىپ باقسا
ئەي قەدىردان ئۇستازلار جاپا تارتساڭلارمۇ ئەۋلادلارنى تەربىيلەشتىن باش تارتماڭلار
ئوبدان قىپتۇ ئۆزىل
ياخشى ساۋات ئىكەن سىزگەكوپتىن كوپ رەھىمەت بۇندىن كىيىن بايقساق بولغۇدەك
پەسىل بېشىدىكى ئۈچ شاھلىق ئۈچۈن ھۇررا
ھاجى ئابدۇغوپۇرغا بارىكاللا !
قەلىمىڭىزگە قابىليىتىڭىزگە ئاپىرىن
بۇ قىزنىڭ ناخشىلىرىنى بەكمۇ ياقتۇرىمەن ئەگەر بۇ خەۋەرنى بالدۇرراق ئاڭلىغان بولسام نەق مەيداندىن كۆرەتتىم
ئەھمىيەتلىك پائالىيەت بۇپتۇ
مۇدىرىچ پېچەت توپتىن بىرنى تەپتى توپ كىردى
مەردلىك سېنى ياخشى كۆرمىگەن كىشىگە ئېھسان قىلىش ۋە سەن ياخشى كۆرمىگەن كىشى بىلەن تاتلىق تىلدا سۆزلىشىشتۇر
ئەمما ما مۇنبەردە يوتكىلىش خەۋەرلىرى قىززىۋاتىدۇ
رەسىم قوشىۋەتسىڭىز بوپتىكەن ئەسلى ياخشى چىقىپتۇ
مىلەنغا كىلىپ خىلە چوڭ بوپ قالدى بۇ ئەقىل چىشى ئەمدى چىقىۋاتقان بولسا كىرەك
مانا ئوغاۋالىدەك ئىش قىپتۇ
ماۋۇ تېمىغا تامغامنى سۇرۇققا تازا چىلاپ مەڭگۈلۈك مۆھۈر بېسىش ئارقىلىق مىننەتدارلىقىمنى ئىپادىلەيمەن
مۇشۇنداق گەپلەرنى قانداق تاپىسىزكىن دەيمەن خېلى ئوخشاپتۇ
ھەممىڭلارغا رەھمەت
ياخشى ئۇقۋېلىشقا تىگىشلىك ئىشكەن
ھەممىسى تەلەيگە باغلىق
بۇ كۈنلەردە مەن بەك تېتىكلىشىپ كېتىۋاتىمەنغۇ ئاللاھقا شۈكرى
توققۇز ئايلانمىدا نۇمۇر تۇلۇق ئېلىپ بۇزۇش تەس بولغان رىكوتتىن بىرنى يارىتىپ قويسىغۇ بەك ياخشى بولاتتى
خۇش كەپسىز خۇشۋاق بولۇڭ
مۆجىزە ياراتقىدەك قانداق كەپشەرلىگەندۇ تېخى مەخسۇس باھالايدىغان جەمىيىتى باركەن
تۇرپاندا ئۈزۈم پىشىپ مەي بوپ كەتتى دەيدىغۇ بۇ كۈنلەرنىمەدىگەن ئېسىل ھە
شۈرلى ياخشى مالتى مو دىگەن ئادەم بىلمىدى قەدىرنى
نىڭ سىناق نەشرى تارقىتىلدى مۇۋەپپەقىيەتلك قاچىلىدىم
ئۇيغۇرۇم ياخشى
نىفىت سىز ھازىر مۇنبەرنىڭ چولپۇنى بولۇپ كەتتىڭىز جۇمۇ رەھمەت تىمىللىرىڭزغا مۇنبەرنى خېلى ئاۋاتلاشتۇرۋاتسىز
ۋەتىنىم كېندىك قىنىم تۈكۈلگەن ئانا تۇپرۇغۇمدۇر ۋەتىنىم ئىللىق قۇچىغىدا ئەللەيلىگەن ئاشۇ خىلۋەت سەھرادۇر
ئورۇنلۇق گەپ قىپتۇ
ئۇھ ئىمتىھاندىن ئۆتۈپتىمەن بەك ياخشى بولدى
خاخا بۇ كارۋاتتا ئۇخلىسا بەكمۇ راھەت ئىكەن
مەن گېرمانىيە ياكى تۈركىيە ئۇيغۇرلېرىنى چېمپيون بولامدىكى دىگەنتىم لېكىن ماۋۇ قازاقىستان ئۇيغۇرلېرىنىڭ ئىچىدە بىرلەشمە ئوينايدىغان ماھىرلارمۇ باركەن ئەمەسمۇ مۇبارەك بولسۇن يەتتە سۇ ئۇيغۇرلېرى
ئۇنداق دېسەكمۇ بولىدىكەن رەخمەت سىزگە !
ئەجەب قىزىقىپ قالدىم
گىرىزمان بولمىسىمۇ بارسىلونا يەنىلا كۈچلۈك
تىرنىر دىگەن ئەنچىلوتىدەك ئىنرىكتەك چوڭ سۈپەت بولسا ئۆزگىمۇ ئۆزگىگىمۇ ياخشى
موڭغۇللار چىنە ئۇسسۇلىنى كارامەت ئويناپ قويىدىكەن
ھىنرىنى قوشوپ قويغان بولسىڭىز تىخمۇ ياخشى بولتتىكەن
ئالدىنقى مەيدان خېلى قىززىغان سۇئارىز توپ ئۇرۇپلا مادرىد خۇددىنى يوقىتىپ قويدى مېھمان ئەمەسمۇ توغرا چۈشۈنەيلى
بارى قۇيۇندەك ئالغا
مەنمۇ يەنىلا رونالدوغا قول قويىمەن
ساپلا داڭلىق ھەكىملەرنىڭ ھەممىنى ئەكىلىشنى پىلانلاپتۇ
ئىينىمىز ياخشى سودىدىن بىرنى قىلۋاپتۇ
ھاھاھاھا تازا قىززىق بوپتۇ
كىچىكىمدا قارا يولغۇندىن چىقىدىغان سېرىق تاۋۇزنى جىق يەيتتىم قۇمدەك ھەم تاتلىق چىقاتتى ئۇ تاۋۇزلار رەڭگى كاۋىدەك ساپسېرىقتى دېسە
ئالىي مەكتەپ چاقىرقىنى تاپشۇرۋالغان ۋاقتىمدا بەكمۇ ھاياجانلىنىپ كەتتىم
ھېنىرىي ئىسپانىيەدىن بىر گۈزەل مودىلنى تېپىۋالدى شوڭا ئاجىراشتى
ۋاھ نېمە دېگەن يېقىملىق ناخشا بۇ ۋاي ۋاي
نىمە دىگەن تىلى تاتلىق يېقىملىق شئېر بۇ ؟
بىزنىڭ ئاتۇشلۇق ئەزمەتلىرىمىزمۇ باركەن ئاللاھ ھەممىسىنىڭ ئىشىنى ئوڭۇشلۇق قىلغاي
ئەسسالامۇ ئەلەيكۇم بۈگۈن نېمە قىلاي ئەتىگەندە ئەڭ ياخشىسى قوغۇنغا سۇ ئاچاي قوغۇن پىشسا يەيمىز ئەمەسمۇ
رونالدوغا ئوڭچىلا قالدىغۇ بۇ مۇكاپات
ۋاھ ئىككى ئاي بولۇپ قالدى ھازىر كۈنىگە ئون مىڭ يەشىكتىن كۆپ سېتىلىۋاتىدۇ
