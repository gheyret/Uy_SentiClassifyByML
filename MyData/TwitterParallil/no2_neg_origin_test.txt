مېنى يېتىم ئوغلاق دېگىنىڭدە خىيالىمغا كىرىپ چىقمىدى چۈنكى مېنىڭ ئاتا-ئانام ھايات بىراق سەن ئۆزۈڭنىڭ قاپقارا تۇل ئىكەنلىكىڭنى ئۇنۇتتۇڭ ئىككى بالاڭنىڭ دادىسى ئاللىقاچان ئۆز مەۋجۇتلۇقىنى يوقىتىپ ئىمان دېسە ئىمانى ئەخلاق دېسە ئەخلاقى يوق مەستلىكتە كۈن ئۆتكۈزىدۇ قېنى دېگىنچۇ ، كىم بىچارىكەن ھامما ؟
بۇ يىللىق ئالاھىدە ئىشتات ئوقۇتقۇچىلىق ئىمتىھانىنىڭ ماتېماتىكا تەقلىدى سوئاللىرىنى تاپماق بەك تەسكە توختىدى مۇشۇ شۇغۇ
ھەجەپ خۇشۇم يوق دېسە مۇشۇ تەنە گەپ قىلغان كىشى بىلەن
ھىھى ، بارغانسېرى ياخشىلانماقتا يوق ، ناچارلىشىپ كىتىپتىغۇ .
تەكەببۇرلانماق ئۆز قەدىرىنى بىلمىگەن ئادەم سىياقىدىن چىققان كىشىلەرنىڭ سۈپىتىدۇر

ئىملارىم خاتا چىقىپ قالدى ، بۇ گۈدۈكنىڭ مەسىلىسىمۇ يا كومپىيوتېرنىڭ مەسىلىسىمۇ بىلمىدىم .
ھەي گۇناھكار بەندىلەر ، نېمىگە تويۇشمايدىغانسەن
ھوي ھوي سىز نېرۋىڭىزدىن ئاداشمىغانسىز !
بۇ جاۋابىڭىز قۇياشنىڭ ئىللىق نۇرىغا ئوخشىدى شۇنداق يېقىشلىق لېكىن بۇ سوئالنىڭ جاۋابى ئەمەس
توردا ھەجەپ ئارامىدا قويمىدى ئادەمنى ئىشىنى قىلماي !

نېمىدىگەن گۈزەل ئەمەس بۇ ئۈرۈمچى پويىز بېكىتى
سۆزلەرنى قانداقمۇ ياسىغان بولاتىكىن تاڭ بەكلا پاساھەتلىك 
تازا نورمال ئەمەس دېسە !
ئىككى كۈندىن بىرى كەيپىياتىم ئانچە ياخشى ئەمەس ، ئەجەبا ئاشۇ بىر ئېغىز گەپ شۇنچە بەك تەسىر قىلغانمىدۇ ؟
نىمانداق ئادەم بىلەن قېيىشىدۇ ما ئاسمان بالىلىرىمىزنىڭ بايرىمىمۇ بايرام بولماي قالدىغۇ بۆگۈن
بىزدەك نامراتلارلا ئۆزئارا لاتارىيەدە مىليون چىقىپ قالسا قانداق قىلاتتىڭ دەپ سوراشقا ئامراق مەن مىليون تاپقاندىن كېيىن جاۋاب بىرەلەيمەن
ئوقۇشمۇ پۈتۈپ قالدى كىنىشكىلارنى تاپشۇرۇپ ئالساقلا مەكتەپكە خۇش دەپ يۇرتقا قايتىمىز بىزنى نېمە قىسمەتلەر كۈتىۋاتقاندۇ ؟

ئانا تىلدا ساۋاتلىق بولالمىغان ئادەمنىڭ ئوي پىكىرىمۇ ، خىيالىمۇ ئادەمنى قىينايدۇ .
ئەگەر مۇئەللىم ئەخلەتنى خالىغان يەرگە تاشلىماڭلار دەپ تەربىيە قىلمىغان بولسا سېنى ئاللىقاچان تاشلىۋەتكەن بولاتتىم
تاملاردا مەھمۇد بوۋام ۋە يۈسۈپ بوۋامنىڭ رەسمىلىرى ئىسىغلىق ئاستىدىكى جۇزىدا خۇشەخوشە رومكا سوقۇشتۇرۇشلار ھەي بىزدىكى ئۆزلۈك تۇيغۇسى قىنى !
نارازىلىق بىلدۈرىمەن مەن ھېچقاچان سۈرخانغا ياردەم قىلمىدىم
مۇشۇنداق قىلىپ گۈزەل ئىسلام دىنىمىزنى خۈنۈكلەشتۈرۈپ ، قورقۇنۇچلۇق قىلىپ كۆرسىتىدۇ .
ھەسەت بىركىشىنىڭ راھىتى ئۇچۇن يەنە بىر كىشىنىڭ ئازابقا سالدۇ
پاراڭ ئارلىقىدا نېمىشقا دۇكىنىدا يېمەكلىكنىڭ ئازلىقنى سورغىنىمدا ئىشەنچىلىك يېمەكلىكنىڭ يوقلىقنى ئېيتتى .
ھازىر بۇ ئادەملەرگە نېمە بولغاندۇ دەيمەن ئاپتوبۇستا ياخشىلىق قىلىپ ئورۇن بېرىپ ئۆزۈمگە كۆڭۈل غەشلىكى تىپىۋالدىم
كىشى نېمىگە ئشەنسە كۆزىدە پەقەت شۇلا بولىدۇ مەيلى ئۇ راست بولسۇن ياكى يالغان
بۇ كۈنلەر بەكمۇ زىرىكشلىك بولۇپ كېتىپ باردغۇ باللا ؟
ئەمدىغۇ ئىشلىتەرمىز ئوغرى نەشرىدىن زىرىكمىدىڭلارمۇ ياكى ھەممىڭلار ھەقىقى نەشرىنى سېتىۋالىدىغان بولۇپ كەتتىڭلارمۇ ؟
جېنىم سۈرىيە سىز نەلەردە قىينىلىپ نەلەردە سەرسانسىز ھە
تولا ئۆزگىرىدىغان ئادەملەردىن ھەزەر ئەيلە دەپ ، بۇنداق كىشلەرنىڭ يا دوست ، يا دۈشمەنلىكىنى پەرق ئىتىپ بولماس .
ئەتگەندىن بىرى كومپيۇتېر ئالدىدا ئولتۇرۇپ ئەجەب چارچاپ كەتتىما بىر دەم ئارام ئالغاچ بۇ گۈدۈكنى چىپ باقاي
ئورنۇمدىن تۇرماي قولۇمنى سۈرۈۋاتسام ئىككى قىزچاق كېلىپ ياردەم لازىممۇ دەيدۇ ياق دېسەم ئۇنىماي تارتىشلاپلا تۇرغۇزىۋىتىپ نىمجان دەپ كەتتى
خالىسىڭىز شەخسىي بلوگدىن بىرنى قۇرۇپ ئەركىن پىكىرلىرىڭىزنى ، قاقشاشلىرىڭىزنى يوللىسىڭىز بولىدۇ .
مەن ھاياتىمدىكى ئاللاھنىڭ سۆزىنى ئاڭلىمىغان ياكى ساۋابلىق ئىشلارنى قىلمىغان ۋاقىتنى ئازغان ۋاقىتم دەپ قارايمەن
خۇدايىم ساقلا ، ئىت-مۈشۈكلىرىمنى ئۇستىراق دوختۇرنىڭ قېشىغا ئاپىرىپ داۋالىتارمەن ، سىزنىڭ قېشىڭىزغا ئاپارسام مۈشۈكۈم قاۋايدىغان ، ئىتىم چاشقان قوغلايدىغان بولۇپ قالار ھەقىچان
يۇرتىمىزنىڭ ئاقىۋىتى ، خەلقىمىزنىڭ ئىستىقبالى توغرىسىدا قايغۇرمىدۇق .
كائىناتنى ياراتقان بىر زات سىزنى مۇھاپىزەت قىلۋاتسا نىمگە قايغۇرسىز ؟
ھەممىسى شۇنىڭغا كۈنۈپ كەتكەن ، ھەممىسى خۇشامەتچى يالغان-ياۋداققا كۆنۈك

بارغانسرى غەملىرىم كۆپىيپ كەتتى مىنىڭ !
