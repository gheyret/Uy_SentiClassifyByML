مۇستاپا كامالدىن كېيىنمۇ دۆلەت يەنىلا دەھرىيلىك يولىدا ماڭدى ، مۇسۇلمانلارنىڭ دىنى پائالىيەتلىرىگە چەك قويۇلدى ، ھاكىمىيەت سەللا ئىسلاملىشىشقا يۈزلەنسە ، ئارمىيە ھاكىمىيەتنى ئاغدۇرۇپ يېڭى ھاكىمىيەت تىكلىدى ، ھەتتا بەزى ئەلباشىلار سۈيقەستكە ئۇچرىغانمۇ بولدى .
مىسىر ئەزەلدىن جۇڭگونىڭ ئايرىلماس بىر قىسمى : بۇ قۇرلارنى ئوقۇۋېتىپ تىترەپ كەتتىم
باغداش مۇنبىرىدە ئىككى مۇناپىق زىيالىينىڭ يېڭى يېزىققا قايتىش ھەققىدىكى بىر تېمىسى بار ئىدى ، تورداشلارنىڭ قارشىلىقى بىلەن ئۆچۈرۈلگەن ئوخشايدۇ .
گۈدۈكتە قاملاشمىغان مۇقام توۋلاپ دىققەتنى جەلىپ قىلىمەن دەپ يۈزىڭىزنى چۈشۈرىۋالماڭ ، قانچىلىك يۈزىڭىز بارلىقىنىغۇ بىلمەيمەن ئەمدى شۇنداقتىمۇ ، ئەسكەرتىپ قوياي قېرىنداشلارچە لاۋزا .
بۇلارنىڭ ھەممىسى بىزنىڭ تېروتورىيە ئېگىلىكىمىزگە ۋە ھوقۇقىمىزغا ئوچۇقتىن ئوچۇق دەخلى تەرۇز قىلغانلىق ، غورۇرىمىزنى دەپسەندە قىلغانلىق .
دوست ۋە دۈشمەننى پەرىق ئېتەلمىگەن ئىنساننىڭ بۇ دۇنيادا ياشىشىنىڭ ئۆزى ئارتۇقچە .
ئويلاپ قالدىم ، ئادەمدە ئىمان بولمىسا ۋىجدانمۇ بولمايدىكەن مۇناپىقلار
چەتنىڭ بولغاچ تولا تور ئۈزلىۋالدىكەن بىر ئۈزۈلسە قايتا چىقىرىشقا توغرا كېلىدىكەن ئۆتكەندە چوڭ مۇستەبىتنىمۇ ئاشۇنداق ئۈزۈلۈپ چىقىرىپ بولالمىغاچقا بەيدۇغا چىقارغانتىم
ئىنشائاللاھ تۇرىۋالىدۇ چۈنكى ئۈمىدسزلىك شەيتاننىڭ ئىشى
قەھرىتاندا پىراۋا ئۈچۈن ئوقۇماق ھەققەتەن تەسكەن ، شۇنۇڭغا چۈشلۇق ئۈنۈم كۆرسەملا بولاتتىغۇ
ئادەم مەغلۇبىيەتكە ئادەتلىنىپ قالىدۇ شۇنىڭ بىلەن مەڭگۈ غەلىبە قىلالمايدۇ غەلىبە دېگىنىمىز مەسئۇلىيەتنى ئادا قىلىش ئىقتىدارىغا ئىگە بولۇش دېگەنلىكتۇر
بىزدە ھېيت بولسا ناغرا-سۇناي ئاۋازى ياڭرايدۇ ، بۇ كىشىنى جۇشقۇنلۇققا باشلايدۇ ، ما خەقتە ھېيت بولسا پوچاڭزا ئاۋازى ، بۇ كىشىگە جەڭ مەيدانىنى ئەسلىتىدىكەن ، ئېتىلىۋاتقان پوجاڭزا ۋە گۈڭگۈرتنىڭ قاڭسىق ھىدى بومبا-ئوقلارغا ئوخشاش كىشىنى سەسكەندۈرىدىكەن
يولۇچىلاردىن ئۆزى ئايرىم يول ئېچىشنى تەلەپ قىلىش بىمەنىلىك يولۇچىلار يولدىن ماڭىدۇ ئەمما يولۇچىنىڭ يول ئېچىش ۋەزىپىسى يوق
بۈگۈنكى كورلا بەك سوغۇق ئۆلمۈدۈم ئەمما ئۇنىڭ بېرىسى بولۇپ كەتتىم تۆنۈگۈن كەچتە ئۇخلىمىغانغا تويدۇم
شەخسى مەنپەتى ۋە ئۆز ھېىسياتنى بىر مىللەتكە بويسۇندۇردىغانلا قانچىكى كۆپەيسە ئۇ جەمىيئەت شۇنچە زاۋاللىققا يۈز تۇتقان بۇلىدۇ .
ئاشۇ تور نامنى قولدىن بېرىپ قويغانغا بەكلا ئۆكۈنىمىز جۇمۇ ؟
ھەقىقەتەن توغرا دەيسىز ھازىر بىزدە ئىلمىي تەپەككۇر بەك كەملەپ كېتىۋاتىدۇ
